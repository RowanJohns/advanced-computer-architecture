LI r0 5        # compute 5!
LI r4 3            # save the return address
fact                   # arg: n in r0, returns n! in r1
    SW r4 0            # store return address in memory
    SW r2 1            # store n in memory
    # base case check
    LI r3 2
    BLT r0 r3 done
    # move n so it is stored next time
    MOVE r2 r0        # save our n
    SUBI r0 1
    LI r4 10        # set return address to 11
    B fact          # call fact (n-1)
    # recursion
    MUL r3 r2 r3  # r3 <- n * fact(n-1)
done
    LW r4 0
    LW r2 1
    JR r4
    HALT
